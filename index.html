#include #include #include #define DHTPIN 4 #define DHTTYPE DHT11 DHT dht(DHTPIN, DHTTYPE); const char* ssid = "YOUR_WIFI_SSID"; const char* password = "YOUR_WIFI_PASSWORD"; #define FIREBASE_HOST "your-project.firebaseio.com" #define FIREBASE_AUTH "your-database-secret" FirebaseData firebaseData; void setup() { Serial.begin(115200); WiFi.begin(ssid, password); Firebase.begin(FIREBASE_HOST, FIREBASE_AUTH); dht.begin(); } void loop() { float temp = dht.readTemperature(); float hum = dht.readHumidity(); Firebase.setFloat(firebaseData, "/smartbin/temperature", temp); Firebase.setFloat(firebaseData, "/smartbin/humidity", hum); delay(5000); }
